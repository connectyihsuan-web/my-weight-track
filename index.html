function renderYTDChart() { const ctx = document.getElementById('ytdChart').getContext('2d'); if (charts.ytd) charts.ytd.destroy(); // 取得所有體重數據，用來在滑動時判斷顯示 const weightData = allData.map(d => d.KG); charts.ytd = new Chart(ctx, { type: 'line', data: { labels: allData.map(d => d.DATE.split('-').slice(1).join('-')), datasets: [{ data: weightData, fill: true, tension: 0.4, borderWidth: 2, pointRadius: 0, // 平時不顯示點 pointHoverRadius: 5, // 滑鼠移過去時放大顯示點，方便對準 pointBackgroundColor: '#8e9fb1', spanGaps: true, segment: { borderColor: ctx => segmentIsPeriod(ctx, allData) ? '#9e827a' : '#8e9fb1', backgroundColor: ctx => segmentIsPeriod(ctx, allData) ? 'rgba(158, 130, 122, 0.15)' : 'rgba(142, 159, 177, 0.05)', } }] }, options: { responsive: true, maintainAspectRatio: false, // 關鍵修改：只要在該日期垂直區域滑動，就會自動抓取最近的點 interaction: { mode: 'index', intersect: false }, plugins: { legend: { display: false }, tooltip: { enabled: true, // 確保提示框開啟 backgroundColor: 'rgba(45, 45, 45, 0.85)', // 莫蘭迪深色背景 titleColor: '#ffffff', bodyColor: '#ffffff', padding: 10, cornerRadius: 8, callbacks: { title: function(context) { return `日期: 2026-${context[0].label.replace('-', '/')}`; }, label: function(context) { const d = allData[context.dataIndex]; let label = ` 體重: ${context.parsed.y} kg`; if (d.備註) { label += ` | 📝 ${d.備註}`; } return label; } } } }, scales: { y: { beginAtZero: false, ticks: { font: { size: 10 } } }, x: { grid: { display: false }, ticks: { font: { size: 10 } } } } } }); }
